class Solution {
  public int[] solution(int[] array) {
    int[] answer = {};
    
    int[] sort_array = Arrays.copyOf(array, array.length);
    
    Arrays.sort(sort_array);
    
    int max = sort_array[sort_array.length - 1];
    
    for (int i = 0; i < array.length; i++) {
      if (array[i] == max) {
        answer = new int[]{ max, i };
      }
    }
    
    return answer;
  }
}
